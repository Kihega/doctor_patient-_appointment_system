sudo apt install mariadb-server -y
sudo service mariadb start

🔹 TASK T2: Secure MariaDB

🔹 TASK T3: Create Database & User
sudo mysql -u root -p

SQL Commands:
CREATE DATABASE appointment_db;

CREATE USER 'appuser'@'localhost' IDENTIFIED BY 'password123';

GRANT ALL PRIVILEGES ON appointment_db.* TO 'appuser'@'localhost';

FLUSH PRIVILEGES;
EXIT;

🔹 TASK T5: Custom User Model (models.py)

from django.contrib.auth.models import AbstractUser
from django.db import models

class User(AbstractUser):
ROLE_CHOICES = (
('admin', 'Admin'),
('doctor', 'Doctor'),
('receptionist', 'Receptionist'),
)
role = models.CharField(max_length=20, choices=ROLE_CHOICES)

---

🔹 TASK T6: Doctor Model

class Doctor(models.Model):
user = models.OneToOneField(User, on_delete=models.CASCADE)
specialization = models.CharField(max_length=100)

def __str__(self):
    return self.user.username

---

🔹 TASK T7: Appointment Model

class Appointment(models.Model):
patient = models.ForeignKey(User, on_delete=models.CASCADE, related_name='appointments')
doctor = models.ForeignKey(Doctor, on_delete=models.CASCADE)
appointment_date = models.DateTimeField()
status = models.CharField(max_length=20, default='scheduled')

def __str__(self):
    return f"{self.patient} - {self.doctor}"

---

🔹 TASK T8: Register Models in Admin
