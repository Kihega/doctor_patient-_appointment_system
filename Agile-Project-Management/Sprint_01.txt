4. SPRINT TASK IMPLEMENTATION (COMMANDS)

---

🔹 Task T1: Update System
sudo apt update && sudo apt upgrade -y

🔹 Task T2: Install Python Tools
sudo apt install python3 python3-pip python3-venv -y

🔹 Task T3: Install MariaDB Dependencies
sudo apt install mariadb-client libmariadb-dev libmariadb-dev-compat -y

🔹 Task T4: Install Build Tools
sudo apt install build-essential -y

🔹 Task T5: Install Node.js
sudo apt install nodejs npm -y

🔹 Task T6: Create Project Folder
mkdir appointment_system
cd appointment_system

🔹 Task T7: Create Virtual Environment
python3 -m venv venv
source venv/bin/activate

🔹 Task T8: Install Backend Dependencies
pip install django djangorestframework mysqlclient

🔹 Task T9: Create Django Project
django-admin startproject backend
cd backend

🔹 Task T10: Run Server
python manage.py runserver 0.0.0.0:8000

🔹 Task T11: Create App
python manage.py startapp appointments

🔹 Task T12: Register App
nano backend/settings.py

Add:
'rest_framework',
'appointments',
